22 public void OnDrag(PointerEventData _eventData)
23 {
24 //Calculate the half size difference between the background and handle rects
25 Vector2 difference = new Vector2
26 {
27 x = (background.rect.size.x - handle.rect.size.x) * .5f,
28 y = (background.rect.size.y - handle.rect.size.y) * .5f,
29 };
30 //Calculate the axis of the input based on the event data and the relative
31 //position to the background's center
32 Axis = Vector2.ClampMagnitude(new Vector2()
33 {
34 x = (_eventData.position.x - background.position.x) / difference.x,
35 y = (_eventData.position.y - background.position.y) / difference.y
36 }, 1);
37 //Apply the axis position to the handles position
38 handle.position = new Vector3
39 {
40 x = (Axis.x * difference.x) + background.position.x,
41 y = (Axis.y * difference.y) + background.position.y
42 };
43 //Apply the deadzone effect after the handle has been placed to prevent
44 //the handle from visually being stuck in the deadzone
45 Axis = (Axis.magnitude < deadzone) ? Vector2.zero : Axis;
46 }
